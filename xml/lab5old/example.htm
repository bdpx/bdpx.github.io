Практический пример: вывод дерева XML

<script>
        function parse (node) {
               if (node.nodeType==1) {
                       document.write ("<ul>"); //сдвигаем дочерние элементы
                       
                       document.write ("<span class='tag'>&lt;"+node.tagName+"</span>");//вывод стартового тэга
                     for (var i=0; i<node.attributes.length; i++) //выводим атрибуты
                               document.write(" <span class='attr'>"+node.attributes.item(i).nodeName+
                                      "</span>=<span class='val'>\""+
                                      node.attributes.item(i).nodeValue+"\"</span>"); 
 
                       if (node.hasChildNodes()) //тэги завершаются иначе у элементов с содержимым
                               document.write("<span class='tag'>&gt;</span>");
                       else  //... и без содержимого
                               document.write("<span class='tag'> /&gt;</span>");
 
                       for (var i=0; i<node.childNodes.length; i++) //перебираем дочерние элементы
                               parse (node.childNodes.item(i)); 
 
                       if (node.hasChildNodes()) //если были дочерние элементы, выводим заверш. тэг
                               document.write ("<span class='tag'>&lt;/"+node.tagName+"&gt;</span>");
                       document.write ("</ul>");
               }
               else if  (node.nodeType==3) //текстовые узлы
                       document.write ("<em>"+node.nodeValue+"</em>");
               else return; //узлы других типов не обрабатываем (комментарии, ссылки на сущности и т.п.)
               
        }
 
        var xmlDoc = new ActiveXObject ("Msxml2.DOMDocument.4.0");
        xmlDoc.async =false;
        xmlDoc.validateOnParse = false;
        xmlDoc.resolveExternals = false;
 
        var xmlFile = prompt ("Input path to an XML file: ", "books.xml")
        xmlDoc.load(xmlFile);
        
        if (xmlDoc.parseError.errorCode != 0) {
               var myErr = xmlDoc.parseError;
               alert("Error: "+myErr.reason+"\nLine: "+myErr.line+
                       "\nPos: "+myErr.linepos+"\nSource code: "+myErr.srcText);
        }
        else {
               var root = xmlDoc.documentElement;            
 
               parse (root);  //начинаем разбор с корневого элемента        
        }
 
</script>

Для обработки дерева в примере используется функция parse. Она вызывает рекурсивно сама себя. Это дает нам возможность обойти все дерево элементов, ввести их сатрибутами и текстом.
